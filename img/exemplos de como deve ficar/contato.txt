<script>
  function dropdownResponsive(elemento) {
    const isActived = $(elemento).attr('aria-expanded');
    if (isActived === 'true') {
       $(elemento).attr('aria-expanded', 'false');
       $(elemento).next(".dropdown-menu").removeClass('collapse-dropdown');
    } else {
       $(elemento).attr('aria-expanded', 'true');
       $(elemento).next(".dropdown-menu").addClass('collapse-dropdown');
    }
  }

		var rgb = [255, 0, 0];
		
		function hexToRgb(hex) {
			var bigint = parseInt(hex, 16);
			var r = (bigint >> 16) & 255;
			var g = (bigint >> 8) & 255;
			var b = bigint & 255;
			rgb[0] = r;
			rgb[1] = g;
			rgb[2] = b;
		}
		function getContrast() {
		  var o = Math.round(((parseInt(rgb[0]) * 299) +
							  (parseInt(rgb[1]) * 587) +
							  (parseInt(rgb[2]) * 114)) / 1000);
		  var fore = (o > 125) ? 'black' : 'white';
		  return fore;
